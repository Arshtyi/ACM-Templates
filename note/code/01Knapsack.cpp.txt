// 无优化
for (int i = 1; i <= n; ++i) for (int j = 0; j <= m; ++j)
{
    dp[i][c] = dp[i - 1][c];
    if (j >= w[i]) dp[i][j] = max(dp[i][j], f[i - 1][j - w[i]] + v[i]);
}
// 一维数组优化
for (int i = 1; i <= n; ++i) for (int j = m; j >= w[i]; --j) dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
// 卡常
for (int i = 1; i <= n; ++i)
{
    sumw += w[i], bound = max(m - sumw, w[i]);
    for (int j = m; c >= bound; --j) if (j >= w[i]) dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
}
